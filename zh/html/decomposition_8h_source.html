173 struct ExpressionDecomposer {
174  // The right operator for capturing expressions is "<=" instead of "<<" (based on the
175  // operator precedence table) but then there will be warnings from GCC about "-Wparentheses"
176  // and since "_Pragma()" is problematic this will stay for now...
177  // https://github.com/catchorg/Catch2/issues/870
178  // https://github.com/catchorg/Catch2/issues/565
179 
180  // For temporary objects, we need to use rvalue reference to avoid copy
181  template <typename L>
182  Expression<L> operator<<(L&& operand) {
183  return Expression<L>(static_cast<L&&>(operand));
184  }
185 
186  // For other objects, we will store the reference
187  template <typename L,
188  typename std::enable_if<!std::is_rvalue_reference<L>::value, void>::type* = nullptr>
189  Expression<const L&> operator<<(const L& operand) {
190  return Expression<const L&>(operand);
191  }
192 };
