171  virtual void RunOneTime(const uint8_t* data, size_t size) override {
172  Base::count++;
173  std::string input = std::string((const char*)data, size);
174  typename Domain::CorpusType corpus = TryParse(input);
175  typename Domain::ValueType value = m_domain.GetValue(corpus);
176  apply(value, detail::gen_seq<std::tuple_size<typename Domain::ValueType>::value>{});

180  unsigned int seed) override {
181  Rng rng(seed);
182  std::string input = std::string((const char*)data, size);
183  typename Domain::CorpusType corpus = TryParse(input);
184  m_domain.Mutate(rng, corpus, false);
185  IRObject mutated_obj = m_domain.SerializeCorpus(corpus);
186  return IRObject::ToString(mutated_obj);
